        function generateUUID() {
            return 'xxxxxxxx-xxxx-4xxx-yxxx-xxxxxxxxxxxx'.replace(/[xy]/g, function(c) {
                const r = Math.random() * 16 | 0, v = c === 'x' ? r : (r & 0x3 | 0x8);

        const modalMessage = document.getElementById('modalMessage');
        let isSignup = true;

        function showModal(message = '') {
            modal.style.display = 'flex';
            modalMessage.textContent = message;
        }

            modal.style.display = 'none';
            modalMessage.textContent = '';
        }

        function toggleAuthMode() {
            isSignup = !isSignup;
            modalTitle.textContent = isSignup ? 'Sign Up' : 'Log In';
            authButton.textContent = isSignup ? 'Sign Up' : 'Log In';
            toggleText.textContent = isSignup ? 'Already have an account? Log In' : 'Need an account? Sign Up';
        }

        // User Management - Place after modal declarations
        let user = JSON.parse(localStorage.getItem('user')) || null;
        if (!user) {
            user = { id: generateUUID(), loggedIn: false };
            localStorage.setItem('user', JSON.stringify(user));
            showModal();
        } else if (!user.loggedIn) {
            showModal();
        }

        function updateSectionVisibility() {
            const sections = ['get-started', 'demo'];
            sections.forEach(sectionId => {

                return;
            }

            if (isSignup) {
                // Check if email is already registered
                if (user && user.email === email) {
                    modalMessage.textContent = 'This email is already signed up. Please log in or use a different email.';
                    return;
                }
                user = { id: generateUUID(), email, password, loggedIn: true };

                if (user.email === email && user.password === password) {
                    user.loggedIn = true;

                    alert('Logged in successfully!');
                } else {
                    modalMessage.textContent = 'Invalid email or password.';
                    return;

                section.style.display = 'none';
